79 void
80 petsciiconv_toascii(char *buf, unsigned int len)
81 {
82  static char c;
83 
84  ptr = buf;
85  for(i = len; i > 0; --i) {
86  c = *ptr;
87  if(c == 0x0a) {
88  c = 0x0d;
89  } else if(c == 0x0d) {
90  c = 0x0a;
91  }
92  if(c != 0x40) {
93  switch (c & 0xe0) {
94  case 0x40:
95  case 0x60:
96  c ^= 0x20;
97  break;
98  case 0xc0:
99  c ^= 0x80;
100  break;
101  }
102  }
103  *ptr = c & 0x7f;
104  ++ptr;
105  }
106 }

114  ++ptr;
115  }
