76 /* Generic MAC return values. */
77 enum {
78  /**< The MAC layer transmission was OK. */
79  MAC_TX_OK,
80 
81  /**< The MAC layer transmission could not be performed due to a
82  collision. */
83  MAC_TX_COLLISION,
84 
85  /**< The MAC layer did not get an acknowledgement for the packet. */
86  MAC_TX_NOACK,
87 
88  /**< The MAC layer deferred the transmission for a later time. */
89  MAC_TX_DEFERRED,
90 
91  /**< The MAC layer transmission could not be performed because of an
92  error. The upper layer may try again later. */
93  MAC_TX_ERR,
94 
95  /**< The MAC layer transmission could not be performed because of a
96  fatal error. The upper layer does not need to try again, as the
97  error will be fatal then as well. */
98  MAC_TX_ERR_FATAL,
99 };
