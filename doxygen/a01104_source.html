98 DESCRIPTION
99 <<vprintf>>, <<vfprintf>>, <<vasprintf>>, <<vsprintf>> and <<vsnprintf>> are
100 (respectively) variants of <<printf>>, <<fprintf>>, <<asprintf>>, <<sprintf>>,
101 and <<snprintf>>. They differ only in allowing their caller to pass the
102 variable argument list as a <<va_list>> object (initialized by <<va_start>>)
103 rather than directly accepting a variable number of arguments.
104 
105 RETURNS
106 The return values are consistent with the corresponding functions:
107 <<vasprintf>>/<<vsprintf>> returns the number of bytes in the output string,
108 save that the concluding <<NULL>> is not counted.
109 <<vprintf>> and <<vfprintf>> return the number of characters transmitted.
110 If an error occurs, <<vprintf>> and <<vfprintf>> return <<EOF>> and
111 <<vasprintf>> returns -1. No error returns occur for <<vsprintf>>.
112 
113 PORTABILITY
114 ANSI C requires all three functions.
115 
116 Supporting OS subroutines required: <<close>>, <<fstat>>, <<isatty>>,
117 <<lseek>>, <<read>>, <<sbrk>>, <<write>>.
118 */

1560 #else

1562  fmt += 1;

1591  break;
1592  case 'L':
1593  flags |= LONGDBL;
1594  break;
1595  case 'q':
1596  flags |= QUADINT;
1597  break;
1598  case 'l':
1599  default:
1600  if (*fmt == 'l')

1603  ++fmt;
1604  }

1718  number = (ch - '0');
1719  while ((ch = *fmt) != '\0' && is_digit(ch))
1720  {

1723  }
1724  break;
1725  case SKIPNUM: /* we have a number to skip */

1733  }

1737  have seen the end of the string, then process up to the max argument needed */

1755  break;

1758  break;
1759  case LONG_DOUBLE:
1760  args[numargs++].val__LONG_DOUBLE = va_arg(*ap, _LONG_DOUBLE);
1761  break;
1762  case WIDE_CHAR:
1763  args[numargs++].val_wint_t = va_arg(*ap, wint_t);
1764  break;
1765  case INT:
1766  case SHORT_INT:
1767  case CHAR:
1768  default:
