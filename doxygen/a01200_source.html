207 /**
208  * \brief Sets \c bit in the \c reg register or byte.
209  * @note Assuming \c reg is an IO register, some platforms
210  * can implement this in a single atomic operation.
211 */
212 #define SETBIT(reg, bit) reg |= BIT(bit)
213 
214 /**
215  * \brief Sets the bits in the \c reg register or the byte
216  * as specified in the bitmask \c bits.
217  * @note This is never a single atomic operation.
218  */
219 #define SETBITS(reg, bits) reg |= (bits)
220 
221 /**
222  * \brief Clears a bit in the \c reg register or byte.
223  * @note Assuming \c reg is an IO register, some platforms (such as the AVR)
224  * can implement this in a single atomic operation.
225  */
226 #define CLEARBIT(reg, bit) reg &= ~(BIT(bit))
227 
228 /**
