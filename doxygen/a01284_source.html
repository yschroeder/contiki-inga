107 {
108  struct httpd_state *s = (struct httpd_state *)state;
109 
110  if(s->file.len > uip_mss()) {

112  } else {
113  s->len = s->file.len;
114  }
115  httpd_fs_cpy(uip_appdata, s->file.data, s->len);

150  //How much to send
151  s->file.len = s->len;
152 
153  do {
154  PSOCK_GENERATOR_SEND(&s->sout, generate, s);
155  s->file.len -= s->len;
156  s->file.data += s->len;
157  } while(s->file.len > 0);
158 
159  s->len = oldlen;
160  s->file.len = oldfilelen;
161  s->file.data = olddata;
162 
163  PSOCK_END(&s->sout);

186 char *
187 get_scriptname(char *dest, char *fromfile)
188 {
189  uint8_t i=0,skip=1;
190  /* Extract a file or cgi name, trim leading spaces and replace termination with zero */
191  /* Returns number of characters processed up to the next non-tab or space */
192  do {
193  dest[i]=httpd_fs_getchar(fromfile++);
194  if (dest[i]==ISO_colon) {if (!skip) break;} //allow leading colons
195  else if (dest[i]==ISO_tab ) {if (skip) continue;else break;}//skip leading tabs
196  else if (dest[i]==ISO_space) {if (skip) continue;else break;}//skip leading spaces
197  else if (dest[i]==ISO_nl ) break; //nl is preferred delimiter
198  else if (dest[i]==ISO_cr ) break; //some editors insert cr
199  else if (dest[i]==0 ) break; //files are terminated with null
200  else skip=0;
201  i++;
202  } while (i<(MAX_SCRIPT_NAME_LENGTH+1));
203  fromfile--;
204  while ((dest[i]==ISO_space) || (dest[i]==ISO_tab)) dest[i]=httpd_fs_getchar(++fromfile);
205  dest[i]=0;
206  return (fromfile);
207 }

210 static
211 PT_THREAD(handle_script(struct httpd_state *s))
212 {
213  /* Note script includes will attach a leading : to the filename and a trailing zero */
214  static char scriptname[MAX_SCRIPT_NAME_LENGTH+1],*pptr;
215  static uint16_t filelength;
216 
217  PT_BEGIN(&s->scriptpt);
218 
219  filelength=s->file.len;
220  while(s->file.len > 0) {
221  /* Sanity check */
222  if (s->file.len > filelength) break;
223 
224  /* Check if we should start executing a script, flagged by %! */
225  if(httpd_fs_getchar(s->file.data) == ISO_percent &&
226  httpd_fs_getchar(s->file.data + 1) == ISO_bang) {
227 
228  /* Extract name, if starts with colon include file else call cgi */
229  s->scriptptr=get_scriptname(scriptname,s->file.data+2);
230  s->scriptlen=s->file.len-(s->scriptptr-s->file.data);
231  PRINTF("httpd: Handle script named %s\n",scriptname);
232  if(scriptname[0] == ISO_colon) {
233  if (httpd_fs_open(&scriptname[1], &s->file)) {
234  PT_WAIT_THREAD(&s->scriptpt, send_file(s));
235  }
236  } else {
237  PT_WAIT_THREAD(&s->scriptpt,httpd_cgi(scriptname)(s, s->scriptptr));
238  }
239  next_scriptstate(s);
240 
241  /* Reset the pointers and continue sending the current file. */
242  s->file.data = s->scriptptr;
243  s->file.len = s->scriptlen;
244  } else {
245 
246  /* Send file up to the next potential script */
247  if(s->file.len > uip_mss()) {
248  s->len = uip_mss();
249  } else {
250  s->len = s->file.len;
251  }
252 
253  if(httpd_fs_getchar(s->file.data) == ISO_percent) {
254  pptr = (char *) httpd_fs_strchr(s->file.data + 1, ISO_percent);
255  } else {
256  pptr = (char *) httpd_fs_strchr(s->file.data, ISO_percent);
257  }
258 
259  if(pptr != NULL && pptr != s->file.data) {
260  s->len = (int)(pptr - s->file.data);
261  if(s->len >= uip_mss()) {
262  s->len = uip_mss();
263  }
264  }
265  PRINTF("httpd: Sending %u bytes from 0x%04x\n",s->file.len,(unsigned int)s->file.data);
266  PT_WAIT_THREAD(&s->scriptpt, send_part_of_file(s));
267  s->file.data += s->len;
268  s->file.len -= s->len;
269  }
270  }
271 
272  PT_END(&s->scriptpt);
273 }

319 static
320 PT_THREAD(send_headers(struct httpd_state *s, const char *statushdr))
321 {
322  char *ptr;
323  PSOCK_BEGIN(&s->sout);
324 
325  PSOCK_GENERATOR_SEND(&s->sout, generate_status, (char *)statushdr);
326 
327  ptr = strrchr(s->filename, ISO_period);
328  if (httpd_strncmp("4", statushdr, 1)==0) {
329  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_htm );
330  } else if(ptr == NULL) {
331  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_bin );
332  } else {
333  ptr++;
334  if(httpd_strncmp(ptr, &httpd_mime_htm[5],3)== 0 ||httpd_strncmp(ptr, &httpd_shtml[1], 4) == 0) {
335  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_htm );
336  } else if(httpd_strcmp(ptr, &httpd_mime_css[5]) == 0) {
337  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_css );
338  } else if(httpd_strcmp(ptr, &httpd_mime_png[6]) == 0) {
339  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_png );
340  } else if(httpd_strcmp(ptr, &httpd_mime_gif[6])== 0) {
341  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_gif );
342  } else if(httpd_strcmp(ptr, httpd_mime_jpg) == 0) {
343  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_jpg );
344  } else {
345  PSOCK_GENERATOR_SEND(&s->sout, generate_header, &httpd_mime_txt);
346  }
347  }
348  PSOCK_END(&s->sout);
349 }

376  PT_WAIT_THREAD(&s->outputpt, send_file(s));

378  }
379  PSOCK_CLOSE(&s->sout);
380  PT_END(&s->outputpt);

385 static
386 PT_THREAD(handle_input(struct httpd_state *s))
387 {
388 
389  PSOCK_BEGIN(&s->sin);
390 
391  PSOCK_READTO(&s->sin, ISO_space);
392 
393  if(httpd_strncmp(s->inputbuf, httpd_get, 4) != 0) {
394  PSOCK_CLOSE_EXIT(&s->sin);
395  }
396  PSOCK_READTO(&s->sin, ISO_space);
397 
398  if(s->inputbuf[0] != ISO_slash) {
399  PSOCK_CLOSE_EXIT(&s->sin);
400  }
401 
402  if(s->inputbuf[1] == ISO_space) {
403  httpd_strcpy(s->filename, httpd_indexfn);
404  } else {
405  s->inputbuf[PSOCK_DATALEN(&s->sin) - 1] = 0;
406  strncpy(s->filename, &s->inputbuf[0], sizeof(s->filename));
407 {
408  /* Look for ?, if found strip file name and send any following text to the LCD */
409  uint8_t i;
410  for (i=0;i<sizeof(s->inputbuf);i++) {
411  if (s->inputbuf[i]=='?') {
412  raven_lcd_show_text(&s->inputbuf[i]);
413  if (i<sizeof(s->filename)) s->filename[i]=0;
414  // s->inputbuf[i]=0; //allow multiple beeps with multiple ?'s
415  }
416  if (s->inputbuf[i]==0) break;
417  }
418 }
419  }
420 
421  webserver_log_file(&uip_conn->ripaddr, s->filename);
422 
423  s->state = STATE_OUTPUT;
424 
425  while(1) {
426  PSOCK_READTO(&s->sin, ISO_nl);
427 
428  if(httpd_strncmp(s->inputbuf, httpd_ref, 8) == 0) {
429  s->inputbuf[PSOCK_DATALEN(&s->sin) - 2] = 0;
430  petsciiconv_topetscii(s->inputbuf, PSOCK_DATALEN(&s->sin) - 2);
431  webserver_log(s->inputbuf);
432  }
433  }
434  PSOCK_END(&s->sin);
435 }

450 void
451 httpd_appcall(void *state)
452 {
453 #if DEBUGLOGIC
454  struct httpd_state *s; //Enter here for debugging with output directed to TCPBUF
455  s = sg = (struct httpd_state *)memb_alloc(&conns);
456  if (1) {
457 #else
458  struct httpd_state *s = (struct httpd_state *)state;
459  if(uip_closed() || uip_aborted() || uip_timedout()) {
460  if(s != NULL) {
461  memb_free(&conns, s);
462  }
463  } else if(uip_connected()) {
464  s = (struct httpd_state *)memb_alloc(&conns);
465  if(s == NULL) {
466  uip_abort();
467  return;
468  }
469 #endif
470  tcp_markconn(uip_conn, s);
471  PSOCK_INIT(&s->sin, (uint8_t *)s->inputbuf, sizeof(s->inputbuf) - 1);
472  PSOCK_INIT(&s->sout, (uint8_t *)s->inputbuf, sizeof(s->inputbuf) - 1);
473  PT_INIT(&s->outputpt);
474  s->state = STATE_WAITING;
475  /* timer_set(&s->timer, CLOCK_SECOND * 100);*/
476  s->timer = 0;
477  handle_connection(s);
478  } else if(s != NULL) {
479  if(uip_poll()) {
480  ++s->timer;
481  if(s->timer >= 20) {
482  uip_abort();
483  memb_free(&conns, s);
484  }
485  } else {
486  s->timer = 0;
487  }
488  handle_connection(s);
489  } else {
490  uip_abort();
491  }
492 }
