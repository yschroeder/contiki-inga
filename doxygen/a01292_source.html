66 void
67 webserver_log_file(uip_ipaddr_t *requester, char *file)
68 {
69  /* Print out IP address of requesting host. */
70 #if LOG_CONF_ENABLED
71 #if UIP_CONF_IPV6
72  char buf[48];
73  uint8_t j;
74  j=httpd_cgi_sprint_ip6((uip_ip6addr_t)*requester, buf);
75  buf[j]=':';buf[j+1]=' ';buf[j+2]=0;
76 #else
77  char buf[20];
78  sprintf(buf, "%d.%d.%d.%d: ", requester->u8[0], requester->u8[1],
79  requester->u8[2], requester->u8[3]);
80 #endif /* UIP_CONF_IPV6 */
81 
82  log_message(buf, file);
83 #endif /* LOG_CONF_ENABLED */
84 
85 }
